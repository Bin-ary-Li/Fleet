138  double score() const {
139  // Compute the score of this node, which is used in sampling down the tree
140  // NOTE: this can't be const because the StreamingStatistics need a mutex
141 
142  if(scoring_type == ScoringType::SAMPLE) {
143  // I will add probability "explore" pseudocounts to my parent's distribution in order to form a prior
144  // please change this parameter name later.
145  if(this->parent != nullptr and uniform() <= explore / (explore + statistics.N)) {
146  return parent->score();
147  }
148  else {
149  std::lock_guard guard(stats_mutex); // lock our stats here
150  return statistics.sample();
151  }
152  }
153  else if(scoring_type == ScoringType::UCBMAX) {
154  // score based on a UCB-inspired score, using the max found so far below this
155  if(statistics.N == 0 || parent == nullptr) return infinity; // preference for unexplored nodes at this level, thus otherwise my parent stats must be >0
156  return statistics.max + explore * sqrt(log(parent->statistics.N+1) / log(1+statistics.N));
157 
158  }
159  else if(scoring_type == ScoringType::MEDIAN) {
160  if(statistics.N == 0 || parent == nullptr) return infinity; // preference for unexplored nodes at this level, thus otherwise my parent stats must be >0
161  return statistics.p_exceeds_median(parent->statistics) + explore * sqrt(log(parent->statistics.N+1) / log(1+statistics.N));
162  }
163  else {
164  assert(false && "Invalide ScoringType specified in MCTS::score");
165  }
